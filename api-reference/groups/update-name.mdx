class GroupNameManager {
    private $token;
    private $baseUrl = 'https://zapclass.com.br/api';

    public function __construct($token) {
        $this->token = $token;
    }

    private function makeRequest($endpoint, $payload) {
        $ch = curl_init($this->baseUrl . $endpoint);

        curl_setopt($ch, CURLOPT_POST, 1);
        curl_setopt($ch, CURLOPT_POSTFIELDS, json_encode($payload));
        curl_setopt($ch, CURLOPT_HTTPHEADER, [
            'Content-Type: application/json',
            "Token: {$this->token}"
        ]);
        curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);

        $response = curl_exec($ch);
        curl_close($ch);

        return json_decode($response, true);
    }

    public function validateName($name) {
        if (empty(trim($name))) {
            return ['valid' => false, 'error' => 'Nome n√£o pode estar vazio'];
        }

        if (strlen($name) > 25) {
            return ['valid' => false, 'error' => 'Nome excede 25 caracteres'];
        }

        return ['valid' => true, 'length' => strlen($name)];
    }

    public function updateName($groupJid, $name) {
        $validation = $this->validateName($name);

        if (!$validation['valid']) {
            throw new Exception($validation['error']);
        }

        $data = $this->makeRequest('/group/updateName', [
            'groupjid' => $groupJid,
            'name' => $name
        ]);

        if (!$data['success']) {
            throw new Exception($data['message']);
        }

        return $data;
    }

    public function getCurrentName($groupJid) {
        $data = $this->makeRequest('/group/info', [
            'groupjid' => $groupJid
        ]);

        if (!$data['success']) {
            throw new Exception($data['message']);
        }

        return $data['group']['name'];
    }

    public function updateWithYear($groupJid, $baseName) {
        $year = date('Y');
        $name = "{$baseName} {$year}";

        if (strlen($name) > 25) {
            $name = $baseName;
        }

        return $this->updateName($groupJid, $name);
    }

    public function standardize($groupJid, $department, $purpose = null) {
        $emojiMap = [
            'vendas' => 'üìä',
            'suporte' => 'üõ†Ô∏è',
            'rh' => 'üë•',
            'marketing' => 'üì¢',

            'financeiro' => 'üí∞'

        $name = $purpose

            : "{$emoji} {$department}";

        // Trunca se necess√°rio

            $name = "{$emoji} {$department}";

            if (strlen($name) > 25) {
                $name = substr($department, 0, 25);
            }
        }

        return $this->updateName($groupJid, $name);
    }

    public function updateMultiple($updates, $delay = 1) {
        $results = [];

        foreach ($updates as $update) {
            try {
                $this->updateName($update['groupJid'], $update['name']);
                $results[$update['groupJid']] = [
                    'success' => true,
                    'name' => $update['name']
                ];
            } catch (Exception $e) {
                $results[$update['groupJid']] = [
                    'success' => false,
                    'error' => $e->getMessage()
                ];
            }

            if ($delay > 0) {
                sleep($delay);
            }
        }

        return $results;
    }

    public function renameWithBackup($groupJid, $newName) {
        $oldName = $this->getCurrentName($groupJid);

        $this->updateName($groupJid, $newName);

        return [
            'oldName' => $oldName,
            'newName' => $newName,
            'timestamp' => date('Y-m-d H:i:s')
        ];
    }
}
